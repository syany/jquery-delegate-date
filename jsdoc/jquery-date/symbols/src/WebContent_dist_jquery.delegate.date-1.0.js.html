528      */
529     addMonth : function(offset) {
530       // 第一引数がなければエラーをスロー
531       if (!!!$.isNumeric(arguments[0])) {
532         throw Error('argument Error.');
533       }
534       var offsetYear;
535       var offsetMonth;
536       var calcMonth;
537       // 加算の場合（+）
538       if (offset >= 0) {
539         offsetYear = Math.floor(offset / _MONTH_BY_YEAR_);
540         offsetMonth = Math.floor(offset % _MONTH_BY_YEAR_);
541         calcMonth = this.getMonth() + offsetMonth;
542         if (calcMonth > 11) {
543           offsetYear += 1;
544           calcMonth = calcMonth - _MONTH_BY_YEAR_;
545         }
546       } else {
547         // 減算の場合（-）
548         offsetYear = Math.ceil(offset / _MONTH_BY_YEAR_);
549         offsetMonth = Math.ceil(offset % _MONTH_BY_YEAR_);
550         calcMonth = this.getMonth() + offsetMonth;
551         if (calcMonth < 0) {
552           offsetYear -= 1;
553           calcMonth = _MONTH_BY_YEAR_ + calcMonth;
554         }
555       }
556       // 月末判定（同月に31日があるかなどを確認する）
557       var targetDate = getTmp$Date_(0).setTime(this.getTime());
558       targetDate.setTime(this.getTime());
559       targetDate.setMonth(calcMonth).setFullYear(targetDate.getFullYear() + offsetYear);
560       if (targetDate.getDay() !== this.getDay()) {
561         // 月末対応（指定月に月末がない場合の対応）
562         this.setDay(1);
563         return this.setMonth(calcMonth).setFullYear(this.getFullYear() + offsetYear).setMonthEnd();
564       } else {
565         // 月末が存在する場合、そのまま設定
566         return this.setMonth(calcMonth).setFullYear(this.getFullYear() + offsetYear);
