1 (function($, window, undefined) {
  2   var DELEGATE_DATAKEY = 'delegateInstance';
  3   /**
  4    * @class 委譲クラス生成クラス。 対象クラスのインスタンスメソッド、クラスメソッド及びプロパティを作成します。
  5    * @constructor
  6    * @extends $
  7    */
  8   $.Delegate = function() {
  9     /** @lends $.Delegate */
 10     if ($.isFunction(this)) {
 11       throw Error('Kill if you didn\'t use the \'new\'.');
 12     }
 13     // 実コンストラクタ
 14     this.constructor_.apply(this, arguments);
 15   };
 16 
 17   // インスタンス基底プロパティ、メソッド群
 18   $.extend($.Delegate.prototype,
 19     {
 20       /** @lends $.Delegate# */
 21       /**
 22        * @type 委譲対象クラス
 23        */
 24       delegateClass : Function,
 25       /**
 26        * $.Delegateクラスオブジェクトを初期化します。 <br>
 27        * ・委譲メソッドを生成します。
 28        */
 29       constructor_ : function() {
 30         // 複数の引数パターンに対応するために、new, bindを使用する
 31         this._DELEGATE_OBJ_ = this.newDelegateInstance.apply(this, arguments || []);
 32         /**
 33          * 委譲メソッドを作成します。
 34          */
 35         // 委譲元Dateのインスタンスメソッド、プロパティのリストを取得し、リスト分繰り返す。
 36         var propertyNames = this.getPropertyNames(this.delegateClass.prototype);
 37         for (var i = 0, p = propertyNames[i]; (p = propertyNames[i]); i++) {
 38           (function(that){
 39             var propertyName = p;
 40             $.Delegate.delegateProperty(that, that._DELEGATE_OBJ_, propertyName);
 41           })(this);
 42         }
 43       },
 44       /**
 45        * 対象のプロパティ名称をリストで返却します。
 46        *
 47        * @param {Object}
 48        *        source
 49        * @MemberOf $.Delegate#
 50        */
 51       getPropertyNames : function(source) {
 52         return (!!Object.getOwnPropertyNames) ? Object.getOwnPropertyNames(source) : [];
 53       },
 54       /**
 55        * 委譲クラスのメソッドを実行する。
 56        * @param {String} method 委譲クラスのメソッド名
 57        * @param {Array} args 委譲メソッドノ］引数配列
 58        */
 59       delegateMethod: function(delegateMethod, args) {
 60         return this._DELEGATE_OBJ_[delegateMethod].apply(this._DELEGATE_OBJ_, args || []);
 61       },
 62       /**
 63        * 委譲クラスをインスタンス化
 64        * @return インスタンス化したオブジェクト
 65        */
 66       newDelegateInstance : function() {
 67         //return new (this.delegateClass.bind.apply(this.delegateClass, [ undefined ].concat([].slice.call(arguments))))();
 68         return (!!arguments[0] && arguments[0] instanceof Array) ?
 69           new (this.delegateClass.bind.apply(this.delegateClass, [ undefined ].concat([].slice.call(arguments[0]))))():
 70           new (this.delegateClass.bind.apply(this.delegateClass, [ undefined ].concat([].slice.call(arguments))))();
 71       }
 72     });
 73 
 74   // インスタンス基底プロパティ、メソッド群
 75   $.extend($.Delegate, {
 76     /**
 77      *
 78      */
 79     newInstance: function(sourceClass) {
 80       return (!!arguments[1] && arguments[1] instanceof Array) ?
 81         new (sourceClass.bind.apply(sourceClass, [ undefined ].concat([].slice.call(arguments[1]))))() :
 82         new (sourceClass.bind.apply(sourceClass, [].slice.call(arguments)))();
 83     },
 84     /** @lends $.Delegate */
 85     /**
 86      * $.Delegateのインスタンスを返却する。 スタティック領域に保存し、１リクエスト中のスコープはシングルトンとなる。
 87      *
 88      * @MemberOf $.Delegate
 89      */
 90     getInstance : function() {
 91       var $delegate = $.data(this, DELEGATE_DATAKEY);
 92       if (!!$delegate) {
 93         return $delegate;
 94       }
 95       // 複数の引数パターンに対応するために、new, bindを使用する
 96       $delegate = $.Delegate.newInstance.apply($.Delegate.newInstance, [ this ].concat([].slice.call(arguments)));
 97       //$delegate = $.Delegate.newInstance([ this ].concat([].slice.call(arguments)));
 98 
 99       $.data(this, DELEGATE_DATAKEY, $delegate);
100       return $delegate;
101     },
102     /**
103      * スタティック領域に保存した$.Delegateインスタンスを破棄する（※再作成するために利用する)
104      */
105     removeInstance : function() {
106       $.removeData(this, DELEGATE_DATAKEY);
107       return this;
108     },
109     /**
110      * 委譲プロパティを設定します
111      *
112      * @param {Object}
113      *        target
114      * @param {Object}
115      *        source
116      * @param {Object}
117      *        propertyName
118      */
119     delegateProperty : function(target, source, propertyName) {
120       if (!!target[propertyName]) {
121         return;
122       }
123       var name = propertyName;
124       // メソッドであればメソッド、プロパティであればプロパティの形式で委譲
125       if ($.isFunction(source[name])) {
126         target[name] = function() {
127           var result = source[name].apply(source, arguments);
128           return /^set/.test(name) ? target : result;
129         };
130       } else {
131         target[name] = source[name];
132       }
133     },
134     /**
135      * 対象のプロパテイファイルを設定
136      *
137      * @param {Object}
138      *        targetClass
139      * @param {Object}
140      *        sourceClass
141      */
142     delegateClassProparties : function(targetClass, sourceClass) {
143       // クラスプロパティを設定します
144       var classPropertyNames =
145         (!!targetClass.prototype.getPropertyNames) ? targetClass.prototype.getPropertyNames(sourceClass) : $.Delegate.prototype
146           .getPropertyNames(sourceClass);
147       for (var i = 0, cp = classPropertyNames[i]; (cp = classPropertyNames[i]); i++) {
148         (function(thisClass) {
149           var classPropertyName = cp;
150           // 対象プロパティには設定しない
151           if (/^(?:arguments|caller|prototype)$/.test(classPropertyName)) {
152             return;
153           }
154           // 設定済みには対応しない
155           if (!!thisClass[classPropertyName]) {
156             return;
157           }
158           // プロパティ設定メイン
159           $.Delegate.delegateProperty(thisClass, sourceClass, classPropertyName);
160         })(targetClass);
161       }
162     },
163     /**
164      * 第2引数のクラスで委譲した第一引数のクラスを返す。
165      *
166      * @param {Object}
167      *        targetClass
168      * @param {Object}
169      *        originalClass
170      */
171     inheritInstance : function(originalClass, targetClass) {
172       var T = targetClass || Function;
173       var O = originalClass;
174       var F = function() {
175         if ($.isFunction(this)) {
176           return O.apply(O, arguments);
177         }
178         $.Delegate.apply(this, arguments);
179         T.apply(this, arguments);
180       };
181 
182       // 通常継承
183       F.prototype = Object.create($.Delegate.prototype);
184       // jQuery的継承
185       $.extend(F, $.Delegate);
186       $.extend(F.prototype, {
187         delegateClass : O
188       });
189       // クラス変数、関数追加
190       F.delegateClassProparties(F, O);
191       return F;
192     }
193   });
194 })(jQuery, window);
